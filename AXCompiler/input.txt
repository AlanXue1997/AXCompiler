int a;
int hanoi(int a, int b, int c, int n) {
	if (1 < n) {
		hanoi(a, c, b, n - 1);
		printf("%d->%d\n", a, c);
		hanoi(b, a, c, n - 1);
	}
	else {
		printf("%d->%d\n", a, c);
	}
}
int main() {
	hanoi(1, 2, 3, 2);
	return 0;
}
